-p: Показывает разницу внесенную каждым коммитом.
git log -p -2

--stat: Показывает краткую статистику по каждой версии:
Список измененных файлов;
Их количество;
Количество добавленных файлов;
Количество добавленных и удаленных строк;
Сводная информация.
git log --stat

--pretty: Меняет формат вывода информации.
oneline
short
full
fuller

git log --pretty=oneline

--pretty:format Позволяет выводить записи журнала в выбранном формате.
git log --pretty=format:"%h - %an, %ar : %s",
%H Хеш-код коммита
%h Сокращенный хеш-код коммита
%T Хеш-код дерева
%t Сокращенный хеш-код дерева
%P Хеш-код родительских коммитов
%p Сокращенный хеш-код родительских коммитов
%an Имя автора
%ae Электронная почта автора
%ad Дата создания оригинала(формат учитывает -date= option)
%ar Дата создания оригинала в относительной форме
%cn Имя создателя версии
%ce Электронная почта создателя версии
%cd Дата создания версии
%cr Дата создания версии в относительном формате
%s Комментарий

--graph Добавляет небольшой ASCII-граф с историей ветвления и слияний:
git log --pretty=format:"%h %s" --graph

git log --shortstat Показывает только строку с изменениями/вставками/удалениями от команды --stat

git log --name-only Показывает список измененных файлов после информации о коммите

git log --name-status Показывает список измененных файлов с информацией о добавлении/изменении/удалении

git log --abbrev-commit Показывает только первые несколько символов контрольной суммы SHA-1 вместо 40

git log --relative-date Показывает дату не в полном, а в относительном формате(например, "2 недели назад"
